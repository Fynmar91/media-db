CREATE TABLE log (     
log_id int NOT NULL AUTO_INCREMENT,     
message varchar(255) NOT NULL,
created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,     
PRIMARY KEY (log_id) );

CREATE TABLE media (     
media_id int NOT NULL AUTO_INCREMENT,     
name varchar(120) NOT NULL,  
altname varchar(120),    
addition varchar(24),
year varchar(12),
type int,
status int,
created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,     
PRIMARY KEY (media_id) );
